CREATE TABLE MEDIDAS
(
  ID_MEDIDA INTEGER PRIMARY KEY,
  DESCRICAO VARCHAR(30) NOT NULL
);

CREATE TABLE CATEGORIAS
(
  ID_CATEGORIA INTEGER PRIMARY KEY,
  DESCRICAO VARCHAR(30) NOT NULL
);

CREATE TABLE ITENS 
(
  ID_ITEM INTEGER PRIMARY KEY,
  NOME_ITEM VARCHAR(30) NOT NULL,
  ID_MEDIDA INTEGER NOT NULL,
  ID_CATEGORIA INTEGER NOT NULL,
  ID_CATEGORIA2 INTEGER,
  ID_CATEGORIA3 INTEGER,
  IS_ATIVO BOOLEAN DEFAULT TRUE,
  CONSTRAINT FK_MEDIDA FOREIGN KEY(ID_MEDIDA) REFERENCES MEDIDAS(ID_MEDIDA),
  CONSTRAINT FK_CATEGORIA FOREIGN KEY(ID_CATEGORIA) REFERENCES CATEGORIAS(ID_CATEGORIA),
  CONSTRAINT FK_CATEGORIA2 FOREIGN KEY(ID_CATEGORIA2) REFERENCES CATEGORIAS(ID_CATEGORIA),
  CONSTRAINT FK_CATEGORIA3 FOREIGN KEY(ID_CATEGORIA3) REFERENCES CATEGORIAS(ID_CATEGORIA)
);

CREATE TABLE OBSERVACOES
(
  ID_OBSERVACAO INTEGER PRIMARY KEY,
  DESCRICAO VARCHAR(50) NOT NULL
);

CREATE TABLE LISTAS
(
  ID_LISTA    INTEGER PRIMARY KEY,
  DESCRICAO   VARCHAR(50),
  DT_CRIACAO  TIMESTAMP DEFAULT NOW(),
  IS_ATIVA    BOOLEAN DEFAULT TRUE
);

CREATE TABLE LISTA_ITENS
(
  ID_LISTA_ITEM  INTEGER PRIMARY KEY,
  ID_LISTA       INTEGER NOT NULL,
  ID_ITEM        INTEGER NOT NULL,
  QUANTIDADE     NUMERIC NOT NULL,
  ID_MEDIDA      INTEGER NOT NULL,
  ID_OBSERVACAO  INTEGER,
  IS_MARCADO     BOOLEAN DEFAULT FALSE,  -- item já foi pego no mercado?

  CONSTRAINT FK_LISTA      FOREIGN KEY(ID_LISTA)       REFERENCES LISTAS(ID_LISTA),
  CONSTRAINT FK_ITEM       FOREIGN KEY(ID_ITEM)        REFERENCES ITENS(ID_ITEM),
  CONSTRAINT FK_MEDIDA_LI  FOREIGN KEY(ID_MEDIDA)      REFERENCES MEDIDAS(ID_MEDIDA),
  CONSTRAINT FK_OBSERVACAO FOREIGN KEY(ID_OBSERVACAO)  REFERENCES OBSERVACOES(ID_OBSERVACAO)
);

CREATE SEQUENCE ITENS_SEQ START 1 OWNED BY ITENS.ID_ITEM;
CREATE SEQUENCE MEDIDAS_SEQ START WITH 1 OWNED BY MEDIDAS.ID_MEDIDA;
CREATE SEQUENCE CATEGORIAS_SEQ START WITH 1 OWNED BY CATEGORIAS.ID_CATEGORIA;
CREATE SEQUENCE OBSERVACOES_SEQ START WITH 1 OWNED BY OBSERVACOES.ID_OBSERVACAO;
CREATE SEQUENCE LISTAS_SEQ START WITH 1 OWNED BY LISTAS.ID_LISTA;
CREATE SEQUENCE LISTA_ITENS_SEQ START WITH 1 OWNED BY LISTA_ITENS.ID_LISTA_ITEM;
